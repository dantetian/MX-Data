StringTable = {}
local this = StringTable
local text = {}








do
	text[0] = "æ— "

end






function StringTable.Set(id,stringPara)
	-- body
	if type(id) ~= "number" or type(stringPara) ~= "string" then
		return
	end

	text[id] = stringPara
end

function StringTable.Get(id)
	-- body
	if not text[id]then
		return ""
	end
	return text[id]
end

return StringTable
