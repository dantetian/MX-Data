--[[
	游戏中所有中文文本显示。
	
	请不要在界面或者编辑器中使用中文文本内容。
]]--
StringTable = {}
local this = StringTable
local text = setmetatable({}, {__index=StringTable, __newindex=function (t, k, v)
	if StringTable[k] then error("replicated string id: " .. tostring(k)) end
	StringTable[k] = v
end})

--
--调用接口
--
function StringTable.Set(id,stringPara)
	-- body
	if type(id) ~= "number" or type(stringPara) ~= "string" then
		return
	end

	text[id] = stringPara
end

function StringTable.Get(id)
	-- body
	if not text[id]then
		return ""
	end
	return text[id]
end

return StringTable
